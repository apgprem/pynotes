class BankAccount:
    def __init__(self, owner, balance):
        self.owner = owner
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount

    def withdraw(self, amount):
        self.balance -= amount

BA = BankAccount("Aswin", balance = 50)

BA.deposit(2000)
BA.withdraw(500)

print(BA.balance)

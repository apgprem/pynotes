# 5. Global Keyword Inside Nested Functions

x = 100

def outer():
    def inner():
        global x
        x = 200

    inner()

outer()
print(x)  # Output: 200

200

# 6. Difference Between global and nonlocal

x = 50

def outer():
    x = 10

    def inner():
        global x
        x = 99

    inner()
    print("Outer x:", x)

outer()
print("Global x:", x)

Outer x: 10
Global x: 99

# 7. Tracking State Using Global Variable

def handle_request():
    global requests
    requests += 1

handle_request()
handle_request()
handle_request()

3

# 8. Global Variable with Conditional Logic

mode = "dark"

def toggle_mode():
    global mode
    if mode == "light":
        mode = "dark"
    else:
        mode = "light"

toggle_mode()
print(mode)  # Output: dark

light

# 9. Avoiding Global Abuse (Best Practice Warning)

config = {"theme": "dark"}

def update_config(new_theme):
    config["theme"] = new_theme  # No global keyword needed

# 10. Best Practice: Encapsulation Instead of Global

class Counter:
    def __init__(self):
        self.value = 1.5

    def increment(self):
        self.value += 1

counter = Counter()
counter.increment()
print(counter.value)  # Output: 1

2.5
